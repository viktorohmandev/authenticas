const response = await fetch(
  'https://api.authenticas.com/verifyPurchase',
  {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-API-Key': 'ak_live_your_key_here'
    },
    body: JSON.stringify({
      userId: 'employee-uuid',
      companyId: 'company-uuid',
      amount: 149.99
    })
  }
);

const result = await response.json();

if (result.success) {
  // Purchase approved!
  console.log('Remaining budget:', result.data.remainingBudget);
} else {
  // Purchase denied
  console.log('Reason:', result.data.reason);
}
import requests

response = requests.post(
    'https://api.authenticas.com/verifyPurchase',
    headers={
        'Content-Type': 'application/json',
        'X-API-Key': 'ak_live_your_key_here'
    },
    json={
        'userId': 'employee-uuid',
        'companyId': 'company-uuid',
        'amount': 149.99
    }
)

result = response.json()

if result['success']:
    print(f"Approved! Remaining: ${result['data']['remainingBudget']}")
else:
    print(f"Denied: {result['data']['reason']}")
